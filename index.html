Quick Setup:
  1. Create a new Google Sheet
  2. Add headers in Row 1: Timestamp, Student, Age, Grade, Teacher, Period, Game, Score, Difficulty, Duration, Integrity
  3. Go to Extensions → Apps Script
  4. Replace the code with the script below, then Deploy → New deployment → Web app (set access to "Anyone")

function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); sheet.appendRow([ data.timestamp, data.student, data.age, data.grade, data.teacher, data.period, data.game, data.score, data.difficulty, data.duration, data.integrity ]); return ContentService .createTextOutput(JSON.stringify({status: 'ok'})) .setMimeType(ContentService.MimeType.JSON); }
